<!-- Validation dispatch pattern from RESEARCH.md Pattern 2 -->
```python
def validate_file(path: Path, adapters: dict[str, PlatformAdapter],
                  platform_override: str | None = None) -> list[Violation]:
    pure = PurePath(path)
    if platform_override:
        matching = [adapters[platform_override]]
    else:
        matching = [a for a in adapters.values() if matches_file(a, pure)]
    if not matching:
        return []
    violations = []
    # AS-series fires ONCE per file — not once per adapter
    if is_skill_md(path):
        frontmatter, body_lines = parse_skill_md(path)
        violations.extend(run_as_series(path, frontmatter, body_lines))
    for adapter in matching:
        violations.extend(run_platform_checks(path, adapter))
    return violations
```
